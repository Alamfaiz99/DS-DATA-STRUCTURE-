//implementing recursion
#include<stdio.h>
int fact(int n)
{
    if(n==0)
    {
        
        return 1;
    }
    else 
    {
        return n*fact(n-1);
        
    }
    
    
}
void main()
{
    int num;
    long double fac;
    printf("Enter the number:\n");
    scanf("%d",&num);
    fac=fact(num);
    printf("The factorial of a number is %Lf\n",fac);
    
}
