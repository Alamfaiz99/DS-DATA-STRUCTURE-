#include<stdio.h>
#include<stdlib.h>
struct node
{
    
    int data;
    struct node *next;
    
    
};

int main()
{
    struct node *head,*p,*q,*j;
    int n,i;
    head=(struct node*)malloc(sizeof(struct node));
    printf("Enter the value to be stored:\n");
    scanf("%d",&head->data);
    head->next=NULL;
    q=head;
    j=head;
    printf("Enter the size of list.\n");
    scanf("%d",&n);
    i=1;
    while(i<n)
    {
        p=(struct node*)malloc(sizeof(struct node));
        printf("Enter the value to be stored:\n");
        scanf("%d",&p->data);
        p->next=NULL;
        q->next=p;
        q=p;
        i++;
        
    }
    printf("The entered values in nodes:\n");
    
    while(j!=NULL)
    {
        printf("%d-",j->data);
        j=j->next;
        
        
    }
 return 0;   
    
}
